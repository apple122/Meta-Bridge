# 🛠️ คู่มือการติดตั้ง MetaStock (ฉบับสมบูรณ์)

คู่มือนี้จะช่วยให้คุณติดตั้งโปรเจกต์ MetaStock ตั้งแต่เริ่มต้นจนพร้อมใช้งาน โดยเน้นการตั้งค่า **Supabase** ให้ถูกต้องเพื่อให้ระบบเทรดและ Login ทำงานได้ 100%

---

## 1. การเตรียมความพร้อม (Prerequisites)
ก่อนเริ่มติดตั้ง กรุณาตรวจสอบว่าเครื่องของคุณมี:
- **Node.js**: เวอร์ชัน 18.x หรือสูงกว่า (แนะนำ 20.x+)
- **npm**: เวอร์ชันล่าสุด

---

## 2. การติดตั้งโปรเจกต์ (Frontend Setup)

1. **ติดตั้ง Dependencies**:
   ```bash
   npm install
   ```

2. **สร้างไฟล์ `.env`**:
   คัดลอกไฟล์สร้างไฟล์ใหม่ชื่อ `.env` ที่โฟลเดอร์หลัก (Root) และใส่ค่าดังนี้:
   ```env
   VITE_SUPABASE_URL=https://your-project-url.supabase.co
   VITE_SUPABASE_PUBLISHABLE_KEY=your-anon-key

    # OTP Email Settings (EmailJS)
    VITE_EMAILJS_SERVICE_ID=your_service_id
    VITE_EMAILJS_TEMPLATE_ID=your_otp_template_id
    VITE_EMAILJS_PUBLIC_KEY=your_public_key

    # Specialized Email Templates (Optional - will fallback to template_id)
    VITE_EMAILJS_TEMPLATE_DEPOSIT_ID=your_deposit_template_id
    VITE_EMAILJS_TEMPLATE_WIN_ID=your_win_template_id
    ```
   *(นำค่าเหล่านี้มาจากหัวข้อ API Settings ในหน้า Supabase Dashboard)*

## 3. การตั้งค่า Supabase (Database Setup)

ไปที่ **Supabase Dashboard > SQL Editor** แล้วสร้าง Query ใหม่ จากนั้นคัดลอก SQL ด้านล่างนี้ไปรันตามลำดับครับ:

### 3.1 สร้างตารางพื้นฐาน (Tables)

```sql
-- 1. ตาราง Profiles (ข้อมูลผู้ใช้)
CREATE TABLE profiles (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  username TEXT UNIQUE NOT NULL,
  first_name TEXT,
  last_name TEXT,
  email TEXT UNIQUE NOT NULL,
  password TEXT NOT NULL, -- ระบบนี้ใช้ Plain-text/Custom password
  balance DECIMAL(20, 2) DEFAULT 0,
  is_admin BOOLEAN DEFAULT false,
  is_verified BOOLEAN DEFAULT false,
  otp_code TEXT,
  otp_expires_at TIMESTAMPTZ,
  phone_number TEXT,
  code TEXT, -- รหัสส่วนตัวของผู้ใช้
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- 2. ตาราง Global Settings (ช่องทางติดต่อ)
CREATE TABLE global_settings (
  id TEXT PRIMARY KEY DEFAULT 'main',
  contact_phone TEXT,
  contact_line TEXT,
  contact_telegram TEXT,
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- 3. ตาราง Transactions (ประวัติการเงิน)
CREATE TABLE transactions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  type TEXT CHECK (type IN ('buy', 'sell', 'deposit', 'withdraw', 'win', 'loss')),
  asset_symbol TEXT NOT NULL,
  amount DECIMAL(20, 8) NOT NULL,
  price DECIMAL(20, 8) NOT NULL,
  total DECIMAL(20, 2) NOT NULL,
  status TEXT DEFAULT 'pending',
  binary_type TEXT, -- 'up' หรือ 'down' (สำหรับ Binary)
  binary_result TEXT, -- 'win' หรือ 'loss' (สำหรับ Binary)
  binary_trade_id UUID, -- เชื่อมกับตาราง binary_trades
  created_at TIMESTAMPTZ DEFAULT now()
);

-- 4. ตาราง Binary Trades (ข้อมูลการเทรด Binary)
CREATE TABLE binary_trades (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  type TEXT NOT NULL,
  asset_symbol TEXT NOT NULL,
  amount DECIMAL(20, 2) NOT NULL,
  entry_price DECIMAL(20, 8) NOT NULL,
  payout_percent INTEGER NOT NULL,
  expiry_time BIGINT NOT NULL,
  status TEXT DEFAULT 'pending',
  result_price DECIMAL(20, 8),
  created_at TIMESTAMPTZ DEFAULT now(),
  settled_at TIMESTAMPTZ
);

-- 5. ตาราง Portfolio (สินทรัพย์ที่มีอยู่)
CREATE TABLE portfolio (
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  asset_symbol TEXT NOT NULL,
  units DECIMAL(20, 8) DEFAULT 0,
  updated_at TIMESTAMPTZ DEFAULT now(),
  PRIMARY KEY (user_id, asset_symbol)
);

-- 6. ตารางแจ้งเตือน (Push Subscriptions)
CREATE TABLE push_subscriptions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  subscription JSONB NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now()
);
```

### 3.2 สร้างฟังก์ชันสำหรับการเทรด (RPC Functions)
เพื่อให้การเทรดรันได้โดยไม่มี Error และมีความแม่นยำ (Atomic) ให้รัน SQL นี้ต่อครับ:

```sql
-- ฟังก์ชันสำหรับลงไม้เทรด (Place Trade)
CREATE OR REPLACE FUNCTION place_binary_trade(
  p_user_id UUID,
  p_type TEXT,
  p_asset_symbol TEXT,
  p_amount DECIMAL,
  p_entry_price DECIMAL,
  p_payout_percent INTEGER,
  p_expiry_time BIGINT
) RETURNS JSONB AS $$
DECLARE
  v_current_balance DECIMAL;
  v_trade_id UUID;
BEGIN
  -- 1. เช็คยอดเงิน
  SELECT balance INTO v_current_balance FROM profiles WHERE id = p_user_id FOR UPDATE;
  IF v_current_balance < p_amount THEN
    RETURN jsonb_build_object('success', false, 'message', 'ยอดเงินไม่เพียงพอ');
  END IF;

  -- 2. หักเงิน
  UPDATE profiles SET balance = balance - p_amount WHERE id = p_user_id;

  -- 3. บันทึกไม้เทรด
  INSERT INTO binary_trades (user_id, type, asset_symbol, amount, entry_price, payout_percent, expiry_time, status)
  VALUES (p_user_id, p_type, p_asset_symbol, p_amount, p_entry_price, p_payout_percent, p_expiry_time, 'pending')
  RETURNING id INTO v_trade_id;

  -- 4. บันทึกธุรกรรม (Transaction Log)
  INSERT INTO transactions (user_id, type, asset_symbol, amount, price, total, status, binary_type, binary_trade_id)
  VALUES (p_user_id, 'buy', p_asset_symbol, p_amount / p_entry_price, p_entry_price, p_amount, 'success', p_type, v_trade_id);

  RETURN jsonb_build_object('success', true, 'message', 'ลงไม้เทรดสำเร็จ', 'trade_id', v_trade_id);
END;
$$ LANGUAGE plpgsql;

-- ฟังก์ชันสำหรับตัดสินแพ้ชนะ (Resolve Trade)
CREATE OR REPLACE FUNCTION resolve_binary_trade(
  p_trade_id UUID,
  p_new_status TEXT,
  p_result_price DECIMAL,
  p_payout_amount DECIMAL
) RETURNS JSONB AS $$
DECLARE
  v_user_id UUID;
  v_current_status TEXT;
  v_asset_symbol TEXT;
  v_amount DECIMAL;
  v_binary_type TEXT;
BEGIN
  -- 1. ล็อคและเช็คสถานะปัจจุบัน ป้องกันการรันซ้ำ
  SELECT user_id, status, asset_symbol, amount, type 
  INTO v_user_id, v_current_status, v_asset_symbol, v_amount, v_binary_type 
  FROM binary_trades WHERE id = p_trade_id FOR UPDATE;
  
  IF v_current_status != 'pending' THEN
    RETURN jsonb_build_object('success', false, 'message', 'ไม้นี้ถูกตัดสินไปแล้ว');
  END IF;

  -- 2. อัปเดตสถานะไม้เทรด
  UPDATE binary_trades 
  SET status = p_new_status, result_price = p_result_price, settled_at = now()
  WHERE id = p_trade_id;

  -- 3. ถ้าชนะ ให้บวกเงินคืนเข้า Profile
  IF p_payout_amount > 0 THEN
    UPDATE profiles SET balance = balance + p_payout_amount WHERE id = v_user_id;
  END IF;

  -- 4. บันทึกธุรกรรมผลลัพธ์ (Transaction Log)
  INSERT INTO transactions (user_id, type, asset_symbol, amount, price, total, status, binary_type, binary_result, binary_trade_id)
  VALUES (
    v_user_id, 
    CASE WHEN p_new_status = 'won' THEN 'win' ELSE 'loss' END, 
    v_asset_symbol, 
    v_amount, 
    p_result_price, 
    p_payout_amount, 
    'success', 
    v_binary_type, 
    p_new_status, 
    p_trade_id
  );

  RETURN jsonb_build_object('success', true, 'message', 'ตัดสินผลสำเร็จ');
END;
$$ LANGUAGE plpgsql;
```

## 4. ข้อมูลเริ่มต้น (Seed Data)
แนะนำให้รันเพื่อสร้างข้อมูล Admin และช่องทางติดต่อเริ่มต้น:

```sql
-- สร้างการตั้งค่าเริ่มต้น (ถ้ายังไม่มี)
INSERT INTO global_settings (id, contact_phone, contact_line, contact_telegram)
VALUES ('main', '0812345678', '@yourline', '@yourtelegram')
ON CONFLICT (id) DO NOTHING;

-- สร้าง Admin เริ่มต้น (รหัสผ่านคือ admin123)
INSERT INTO profiles (username, first_name, email, password, is_admin, balance)
VALUES ('admin', 'System', 'admin@metastock.com', 'admin123', true, 10000)
ON CONFLICT (username) DO NOTHING;
```

## 5. การตั้งค่า RLS (Row Level Security)
เพื่อให้ระบบดึงข้อมูลได้โดยไม่ติด Error 403 ให้ไปที่ **Supabase > Settings > Database** และตั้งค่าตามนี้:
1. สำหรับช่วงเริ่มต้น แนะนำให้ **ปิด (Disable) RLS** ในตารางทุกอัน เพื่อความสะดวกในการพัฒนาระบบ
2. หรือหากต้องการเปิด ต้องสร้าง Policy "Enable read/write for service role" หรือ "Enable for all" ครับ

## 7. การตั้งค่าระบบ OTP (EmailJS Setup)

ระบบใช้ **EmailJS** ในการส่งรหัส OTP เพื่อยืนยันตัวตนและการลืมรหัสผ่าน:

1.  **สร้าง Email Template**: ใน EmailJS Dashboard ให้สร้างเทมเพลตใหม่ (แนะนำให้แยกเทมเพลตสำหรับ OTP, เติมเงิน และชนะผลการเทรด)
2.  **ตั้งค่าตัวแปร (Dynamic Variables)**: ในเทมเพลตต้องประกอบด้วยตัวแปรดังนี้เพื่อให้ระบบส่งข้อมูลได้ครบถ้วน:
    - `{{email_subject}}`: หัวข้อเมล (เช่น ยืนยันตัวตน, เปลี่ยนรหัสผ่าน, แจ้งถอนเงิน)
    - `{{email_title}}`: หัวข้อพาดหัวในตัวเมล (Title)
    - `{{to_name}}`: ชื่อผู้รับ
    - `{{passcode}}`: รหัส OTP 6 หลัก หรือ ยอดเงิน (กรณีแจ้งเติมเงิน/ชนะ)
    - `{{email_greeting}}`: คำทักทาย (เช่น สวัสดีครับคุณ)
    - `{{email_body}}`: เนื้อหาหลักของเมล (ซึ่งจะถูกแปลภาษามาให้จากระบบแล้ว)
    - `{{email_otp_label}}`: ข้อความกำกับรหัส (เช่น "รหัสยืนยันของคุณคือ")
    - `{{time}}`: เวลาที่ทำรายการ หรือเวลาหมดอายุ
    - `{{email_expiry}}`: ข้อความแจ้งเตือนหมดอายุหรือคำแนะนำด้านความปลอดภัย
    - `{{email_warning}}`: ข้อความเตือน (เช่น "ห้ามบอกรหัสให้ผู้อื่น")
    - `{{email_footer_brand}}`: ชื่อแบรนด์ที่ท้ายเมล (Meta Bridge)
    - `{{email_footer_auto}}`: ข้อความแจ้งว่าเป็นเมลอัตโนมัติ
3.  **Subject Line**: ในส่วนของหัวข้อ (Subject) ของหน้า EmailJS ให้ใส่ตัวแปร `{{email_subject}}` ไว้เพียงอย่างเดียว เพื่อให้ระบบสามารถส่งหัวข้อภาษาไทย/อังกฤษตามสถานการณ์ได้อัตโนมัติครับ

## 8. การแก้ไขปัญหาเบื้องต้น (Troubleshooting)

### 8.1 บัญชีสร้างใหม่กลายเป็น Admin เสมอ
หากพบว่าผู้ใช้ที่สมัครสมาชิกใหม่ได้รับสิทธิ์ Admin ทันที (is_admin เป็น true) ให้รัน SQL นี้ใน SQL Editor เพื่อแก้ค่า Default ของฐานข้อมูล:
```sql
ALTER TABLE public.profiles 
ALTER COLUMN is_admin SET DEFAULT false;
```

### 8.2 ล็อคอินแล้วค้างที่หน้าโหลด
- ตรวจสอบไฟล์ `.env` ว่าใส่ URL และ API Key ของ Supabase ถูกต้องหรือไม่
- ตรวจสอบว่าได้รัน SQL สร้างตาราง `profiles` ครบถ้วนแล้วหรือไม่
